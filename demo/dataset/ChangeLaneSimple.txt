public class ChangeLaneSimple {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter safe distance: ");
        double safeDist = sc.nextDouble();

        int currentLane = 0;      
        int targetLane  = 1;      

        double carPos   = 80.0;
        double carSpeed = 28.0;

        double aheadPos    = 120.0;
        double aheadSpeed  = 30.0;
        double behindPos   = 40.0;
        double behindSpeed = 25.0;

        String decision = "Stay no change";

        if (currentLane != targetLane) {
            boolean safeAhead  = (aheadPos - carPos)  >= safeDist && aheadSpeed >= carSpeed;
            boolean safeBehind = (carPos - behindPos) >= safeDist && carSpeed  >= behindSpeed;
            if (safeAhead && safeBehind) {
                decision = "Change lane";
            }
        }

        System.out.println("Decision: " + decision);
        sc.close();
    }
}
